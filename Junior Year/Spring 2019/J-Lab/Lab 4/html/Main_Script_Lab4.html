Constant = 1;

    Damping_Ratio_Small_Flood(n) = (2*pi)/Period_Small;
end
Damp_Mean_Small = mean(Damping_Ratio_Small_Flood);

    Damp_Small_Valve(n-1) = Num./sqrt(4*pi^2+Num.^2);

    Damp_Medium_Valve(n-1) = Num/(sqrt(4*pi^2+Num^2));
